import React from 'react';
import PropTypes from 'prop-types';
 
import {ENVDefaultObj} from './dataObj.js';
import {camelCaseTransformerFn} from './applicationFn.js';
import {mainRenderFn} from './mainRenderFn.js';
 
class Menu extends React.Component {
  constructor(props) {
    super(props);
    this.collapseStatusList = {};
  }
  componentWillMount () {
    this.updateENVFn(this.props);
  }
  componentWillReceiveProps(nextProps) {
    this.updateENVFn(nextProps);
  }  
  render () {
    return mainRenderFn(this);
  }
 
  updateENVFn (source) {
    this.env = new ENVDefaultObj().env;
    Object.keys(source).map((entry) => {
      switch (entry)
      {
      case 'styleObj':
        Object.keys(source.styleObj).map((node_entry) => {
          this.env.styleObj[node_entry] = {};
          Object.keys(source.styleObj[node_entry]).map((style_entry) => {
            let camelCaseStyleName = camelCaseTransformerFn(style_entry);
            this.env.styleObj[node_entry][camelCaseStyleName] = source.styleObj[node_entry][style_entry];
          });
        });
        break;
      default:
        this.env[entry] = source[entry];
        break;
      }
    });
  }
}  
 
Menu.propTypes = {
  listArr : PropTypes.array.isRequired
};
 
Menu.defaultProps = {
  listArr : []
};
 
export default Menu;
