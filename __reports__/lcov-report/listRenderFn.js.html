import React from 'react';
 
import {ListInfoObj} from './dataObj.js';
import {createBasicProps} from './applicationFn.js';
 
export function listRenderFn (listThis, source, layerCounter) {
  let infoObj = new ListInfoObj(source);
  let content = [];
  let hasChildren = (0 < infoObj.children.length)? true : false;
  let nextLayer = layerCounter+1;
  let props_item = createBasicProps(listThis.env, 'layer-item');
  let props_content = createBasicProps(listThis.env, 'item-content');
  props_content.onClick = (event) => {_itemOnClickHandler(event, listThis, infoObj);};
  let props_name = createBasicProps(listThis.env, 'content-name');
  content.push(
    <li {...props_item}>
      <div {...props_content}>
        <div {...props_name}>{infoObj.name}</div>
      </div>
      {(() => {
        let content_sublist = [];
        if (hasChildren)
        {
          let props_sublist = createBasicProps(listThis.env, 'item-sublist');
          let props_layer = createBasicProps(listThis.env, 'list-layer');
          let layerNextName = `layer-${nextLayer}`;
          props_layer.className += ` ${layerNextName}`;
          if (listThis.env.styleObj[layerNextName]) {
            Object.keys(listThis.env.styleObj[layerNextName]).map((config) => {
              props_layer.style[config] = listThis.env.styleObj[layerNextName][config];
            });
          }
          content_sublist.push(
            <div {...props_sublist}>
              <ul {...props_layer}>
                {
                  infoObj.children.map((item) => {
                    let content_list = [];
                    content_list.push(listRenderFn(listThis, item, (nextLayer)));
                    return content_list;
                  })
                }
              </ul>
            </div>
          );
        }
        return content_sublist; 
      })()}
    </li>                                       
  ); 
  return content;
}
 
function _itemOnClickHandler (event, listThis, infoObj) {
  event.stopPropagation();
  listThis.env.itemOnClickFn(infoObj);
  listThis.forceUpdate();
}
